// Reçoit la confirmation CinetPay et :
//   • Active l'annonce après paiement publication
//   • Prolonge l'annonce après renouvellement
//   • Retire l'annonce après réservation locataire
// =============================================

const admin = require("firebase-admin");

function initFirebase() {
  if (!admin.apps.length) {
    admin.initializeApp({
      credential: admin.credential.cert({
        projectId:   process.env.FIREBASE_PROJECT_ID,
        clientEmail: process.env.FIREBASE_CLIENT_EMAIL,
        privateKey:  process.env.FIREBASE_PRIVATE_KEY?.replace(/\\n/g, "\n")
      })
    });
  }
  return admin.firestore();
}

export default async function handler(req, res) {
  if (req.method !== "POST") return res.status(405).end();

  const {
    cpm_trans_id,    // ID transaction CinetPay
    cpm_result,      // "00" = succès
    cpm_amount,      // montant payé
    cpm_custom       // notre metadata JSON
  } = req.body;

  // Ignorer les paiements non confirmés
  if (cpm_result !== "00") {
    console.log("Paiement non confirmé:", cpm_trans_id, "code:", cpm_result);
    return res.status(200).json({ message: "ignored" });
  }

  let metadata = {};
  try { metadata = JSON.parse(cpm_custom || "{}"); } catch {}

  const { annonceId, typePaiement, pays } = metadata;
  if (!annonceId || !typePaiement) {
    console.error("Metadata manquante:", cpm_custom);
    return res.status(400).json({ message: "metadata invalide" });
  }

  try {
    const db = initFirebase();
    const annonceRef = db.collection("annonces").doc(annonceId);
    const FieldValue = admin.firestore.FieldValue;
    const Timestamp  = admin.firestore.Timestamp;

    // ── Cas 1 : Logeur paie pour PUBLIER une annonce ──
    if (typePaiement === "publication") {
      await annonceRef.update({
        statut:         "actif",
        dateActivation: FieldValue.serverTimestamp(),
        dateExpiration: Timestamp.fromDate(
          new Date(Date.now() + 30 * 24 * 60 * 60 * 1000)
        ),
        paiementId:     cpm_trans_id,
        smsFailed7j:    false   // flag pour éviter doublon SMS
      });
      console.log("✅ Annonce activée:", annonceId);
    }

    // ── Cas 2 : Logeur RENOUVELLE son abonnement ──
    else if (typePaiement === "renouvellement") {
      const snap = await annonceRef.get();
      const ancienne = snap.data()?.dateExpiration?.toDate() || new Date();
      // Prolonger depuis la date d'expiration actuelle (pas depuis aujourd'hui)
      const nouvelleExp = new Date(ancienne);
      nouvelleExp.setDate(nouvelleExp.getDate() + 30);
      await annonceRef.update({
        statut:           "actif",
        dateExpiration:   Timestamp.fromDate(nouvelleExp),
        dateProlongation: FieldValue.serverTimestamp(),
        smsFailed7j:      false
      });
      console.log("✅ Annonce prolongée:", annonceId, "→", nouvelleExp.toISOString().split('T')[0]);
    }

    // ── Cas 3 : Locataire RÉSERVE → retrait immédiat ──
    else if (typePaiement === "reservation") {
      await annonceRef.update({
        statut:      "loue",
        dateRetrait: FieldValue.serverTimestamp(),
        paiementId:  cpm_trans_id
      });
      console.log("🏠 Annonce retirée (louée):", annonceId);
    }

    // Enregistrer le paiement dans la collection paiements
    await db.collection("paiements").add({
      transactionId:    cpm_trans_id,
      annonceId,
      typePaiement,
      montant:          parseInt(cpm_amount),
      statut:           "success",
      pays,
      dateConfirmation: FieldValue.serverTimestamp()
    });

    return res.status(200).json({ message: "OK" });

  } catch (err) {
    console.error("Erreur webhook:", err);
    return res.status(500).json({ message: "Erreur serveur" });
  }
}
